CHARACTER SET utf8mb4
COLLATE utf8mb4_general_ci;

CREATE TABLE krafne (
  id INT AUTO_INCREMENT PRIMARY KEY,
  ime VARCHAR(50) NOT NULL,
  cijena DECIMAL(4,2) NOT NULL,
  nadjev VARCHAR(50),
  slika VARCHAR(100)
);
